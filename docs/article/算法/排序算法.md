## 冒泡排序 :bulb:

- 比相邻的元素大就交换位置，一轮下来最大的就会到数组到最后，重复这一动作知道排序结束，类似气泡上浮。

```js
function bubbleSort(arr) {
  for (let i = 0; i < arr.length; i++) {
    for (let j = 0; j < arr.length - 1; j++) {
      if (arr[j] > arr[j + 1]) {
        let temp = arr[j];
        arr[j] = arr[j + 1];
        arr[j + 1] = temp;
      }
    }
  }
  return arr;
}
```

## 插入排序 :bulb:

- 从第一个元素开始，认为第一个元素是有序的
- 取下一个元素，和之前的已排好的元素比较，插入到之前的已排好的元素队列中
- 重复此过程直到排序结束

```js
function insertSort(arr) {
  for (let i = 1; i < arr.length; i++) {
    let current = arr[i];
    let tempIndedx = i - 1;
    while (tempIndedx >= 0 && arr[tempIndedx] > current) {
      arr[tempIndedx + 1] = arr[tempIndedx];
      tempIndex--;
    }
    arr[tempIndex + 1] = current;
  }

  return arr;
}
```

## 选择排序 :bulb:

- 在无序的队列中找出最大或者最小的，放到排序序列的起始位置，然后再从剩余的未排序队列中找出最大或者最小、交换位置直到排序结束

```js
function selectSort(arr) {
  let tempIndex = 0;
  for (let i = 0; i < arr.length; i++) {
    tempIndex = i;
    for (let j = i; j < arr.length; j++) {
      if (arr[i] > arr[tempIndex]) {
        tempIndex = i;
      }
    }
    if (tempIndex !== i) {
      let temp = arr[i];
      arr[i] = arr[tempIndex];
      arr[tempIndex] = temp;
    }
  }
  return arr;
}
```

## 归并排序 :bulb:

- 采用分治思想，把长度为 n 的序列分成 2 个长度为 n/2 的子序列，在分别对子序列进行排序

```js
function mergeSort(arr) {
  if (arr.length < 2) {
    return arr;
  }
  let middle = Math.floor(arr.length / 2),
    left = arr.slice(0, middle),
    right = arr.slice(middle);
  return merge(mergeSort(left), mergeSort(right));
}

function merge(left, right) {
  let result = [];
  while (left.length > 0 && right.length > 0) {
    if (left[0] <= right[0]) {
      result.push(left.shift());
    } else {
      result.push(right.shift());
    }
  }
  while (left.length) {
    result.push(left.shift());
  }
  while (right.length) {
    result.push(right.shift());
  }
  return result;
}
```

## 快速排序 :bulb:

- 选择一个元素，把所有比它大的元素放到右边，所有比它小的元素放到左边，在分别对两边对元素进行递归排序直到数组长度小于等于 1 就排序结束

```js
function quickSort(arr) {
  if (arr.length <= 1) {
    return arr;
  }
  let index = 0;
  let pivot = arr.splice(index, 1)[0];

  let left = [];
  let right = [];
  for (let i = 0; i < arr.length; i++) {
    if (arr[i] < pivot) {
      left.push(arr[i]);
    } else {
      right.push(arr[i]);
    }
  }

  return quickSort(left).concat(pivot, quickSort(right));
}
```

```js
function quickSort(arr, left, right) {
  let len = arr.length
  left = left || 0
  right = right || len - 1
  if (left < right) {
    let partitionIndex = partition(arr, left, right)
    quickSort(arr, left, partitionIndex - 1)
    quickSort(arr, partitionIndex + 1, right)
  }
  return arr
}

function partition(arr, left, right) {
  let pivotkey = arr[left]
  while (left < right) {
    while (left < right && arr[right] >= pivotkey) {
      right--
    }
    arr[left] = arr[right]
    while (left < right && arr[left] <= pivotkey) {
      left++
    }
    arr[right] = arr[left]
  }
  arr[left] = pivotkey
  return left
}
```
